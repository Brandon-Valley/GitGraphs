function v12GetToken() {
  try { return sessionStorage.getItem("gitgraphsToken") || ""; } catch (error) { return ""; }
}

function v12SetToken(value) {
  try {
    if (value) sessionStorage.setItem("gitgraphsToken", value);
    else sessionStorage.removeItem("gitgraphsToken");
  } catch (error) {}
  E.tokenButton.textContent = value ? "Token active" : "Use token (faster)";
}

v12SetToken(v12GetToken());
E.tokenButton.onclick = () => {
  const current = v12GetToken();
  const value = prompt(current ? "A token is active. Paste a replacement token, leave blank to remove it, or press Cancel to keep it." : "Optional: paste a GitHub personal access token. It stays in this browser session and is sent only to api.github.com. Leave blank to cancel.", current);
  if (value === null) return;
  v12SetToken(value.trim());
  toast(value.trim() ? "GitHub token enabled for faster full-history loading." : "GitHub token removed.");
};
E.stopHistory.onclick = () => {
  S.fullHistoryCancelled = true;
  E.historyDetail.textContent = "Stopping after the current GitHub request...";
};

function v12IsoDay(milliseconds) {
  return new Date(milliseconds).toISOString().slice(0, 10);
}

function v12WindowDays(startDay, endDay) {
  return Math.floor((endDay - startDay) / DAY) + 1;
}

let v12LastSearchAt = 0;
async function v12Delay(milliseconds) {
  if (milliseconds > 0) await new Promise(resolve => setTimeout(resolve, milliseconds));
}

async function v12AuthenticatedRequest(path) {
  const token = v12GetToken();
  if (!token) return jsonp(path);
  try {
    const response = await fetch(API + path, {
      method: "GET",
      mode: "cors",
      credentials: "omit",
      cache: "no-store",
      headers: {
        Accept: "application/vnd.github+json",
        Authorization: "Bearer " + token
      }
    });
    const text = await response.text();
    let body = null;
    try { body = text ? JSON.parse(text) : null; }
    catch (error) { throw new Error("GitHub returned an unreadable response."); }
    if (!response.ok) {
      const requestError = new Error(body?.message || ("GitHub returned " + response.status));
      requestError.status = response.status;
      requestError.retryAfter = Number(response.headers.get("retry-after") || 0);
      requestError.resetAt = Number(response.headers.get("x-ratelimit-reset") || 0);
      throw requestError;
    }
    window.gitGraphsWorkingTransport = "authenticated";
    return body;
  } catch (error) {
    if (error.status === 401) {
      v12SetToken("");
      toast("That GitHub token was rejected, so GitGraphs switched back to public loading.", true);
    }
    if (error.status && error.status !== 403 && error.status !== 429) throw error;
    return jsonp(path);
  }
}

async function v12SearchRequest(path) {
  const minimumGap = v12GetToken() ? 350 : 6500;
  const wait = minimumGap - (Date.now() - v12LastSearchAt);
  if (wait > 0) await v12Delay(wait);
  let lastError = null;
  for (let attempt = 0; attempt < 3; attempt += 1) {
    if (S.fullHistoryCancelled) throw new DOMException("Loading cancelled", "AbortError");
    try {
      v12LastSearchAt = Date.now();
      return await v12AuthenticatedRequest(path);
    } catch (error) {
      lastError = error;
      if (error.name === "AbortError") throw error;
      if (error.status && error.status !== 403 && error.status !== 429) throw error;
      const waitSeconds = error.retryAfter || (error.resetAt ? Math.max(1, error.resetAt - Math.floor(Date.now() / 1000)) : (attempt + 1) * 12);
      E.historyDetail.textContent = "GitHub asked GitGraphs to slow down. Retrying in " + waitSeconds + " seconds; " + fmt(S.c.length) + " commits are already preserved.";
      await v12Delay(Math.min(waitSeconds, 75) * 1000);
    }
  }
  throw lastError || new Error("GitHub search did not respond.");
}

function v12SearchPath(username, startDay, endDay, page) {
  const query = "author:" + username + " committer-date:" + v12IsoDay(startDay) + ".." + v12IsoDay(endDay);
  return "/search/commits?q=" + encodeURIComponent(query) + "&sort=committer-date&order=asc&per_page=100&page=" + page;
}
