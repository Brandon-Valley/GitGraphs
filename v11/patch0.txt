user = async function(raw) {
  const u = norm(raw);
  if (!valid(u)) return toast("Enter a valid GitHub username.", true);
  S.u = u;
  S.r = "ALL";
  S.scrollToday = true;
  S.total = 0;
  S.incomplete = false;
  S.partial = false;
  E.user.value = u;
  load(true, "Loading @" + u + "...", 4);
  try {
    S.p = await jsonp("/users/" + encodeURIComponent(u));
    const all = [];
    for (let page = 1; page <= 10; page += 1) {
      load(true, "Loading public commit page " + page + " of up to 10...", 8 + page * 8);
      let data;
      try {
        data = await jsonp("/search/commits?q=" + encodeURIComponent("author:" + u) + "&sort=committer-date&order=desc&per_page=100&page=" + page);
      } catch (error) {
        if (all.length && (error.status === 403 || error.status === 429)) {
          S.partial = true;
          break;
        }
        throw error;
      }
      if (page === 1) {
        S.total = Number(data.total_count || 0);
        S.incomplete = Boolean(data.incomplete_results);
      }
      const items = (data.items || []).filter(item => !item.repository?.private);
      all.push(...items);
      if (items.length < 100 || all.length >= Math.min(S.total || 1000, 1000)) break;
    }
    S.c = uniq(all).sort((a, b) => date(b) - date(a));
    show();
    render();
    location.hash = encodeURIComponent(u);
    load(false);
    if (!S.c.length) toast("No public commits were found.", true);
  } catch (error) {
    load(false);
    toast(error.status === 403 ? "GitHub public API limit reached. Try again later." : error.message || "GitHub request failed.", true);
  }
};

show = function() {
  E.home.classList.add("off");
  E.app.classList.add("on");
  E.back.hidden = false;
  E.avatar.src = S.p.avatar_url;
  E.name.textContent = S.p.name || S.p.login;
  const created = S.p.created_at ? new Date(S.p.created_at).toLocaleDateString(undefined, { year: "numeric", month: "long", day: "numeric" }) : "unknown";
  E.meta.textContent = fmt(S.p.public_repos) + " public repositories / " + fmt(S.p.followers) + " followers / joined " + created;
  E.bio.textContent = S.p.bio || "Public commit activity across GitHub.";
  E.gh.href = S.p.html_url;
  scrollTo(0, 0);
};

function dayStart(value) {
  const d = new Date(value);
  return Date.UTC(d.getUTCFullYear(), d.getUTCMonth(), d.getUTCDate());
};

function accountStart() {
  const created = S.p?.created_at ? dayStart(S.p.created_at) : Math.min(...S.c.map(commit => dayStart(date(commit))), dayStart(Date.now()));
  const days = { "30D": 30, "90D": 90, "1Y": 365 }[S.r];
  return days ? Math.max(created, dayStart(Date.now()) - (days - 1) * DAY) : created;
};

start = function() { return new Date(accountStart()); };
