async function requestGitHub(path) { const response = await fetch('https://api.github.com' + path, { method: 'GET', mode: 'cors', credentials: 'omit', cache: 'no-store' }); const body = await response.json(); if (!response.ok) { const error = new Error(body && body.message ? body.message : 'GitHub returned ' + response.status); error.status = response.status; throw error; } return body; }
